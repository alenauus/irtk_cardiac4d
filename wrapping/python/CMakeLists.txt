IF (PYTHON_INCLUDE_PATH)
  INCLUDE_DIRECTORIES(${PYTHON_INCLUDE_PATH})
ENDIF (PYTHON_INCLUDE_PATH)

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR})

SET(CMAKE_SWIG_FLAGS "")
SET_SOURCE_FILES_PROPERTIES(irtk.i PROPERTIES CPLUSPLUS ON)
SET_SOURCE_FILES_PROPERTIES(irtk.i PROPERTIES SWIG_FLAGS "-modern")
SWIG_ADD_MODULE(irtk python irtk.i)

IF (PYTHON_LIBRARY)
  SWIG_LINK_LIBRARIES(irtk ${PYTHON_LIBRARIES})
ENDIF (PYTHON_LIBRARY)

IF (UNIX)
  ADD_CUSTOM_TARGET(copyirtkpy ALL cp irtk.py ${LIBRARY_OUTPUT_PATH}/irtk.py)
ENDIF (UNIX)

IF (WIN32)
  ADD_CUSTOM_TARGET(copyirtkpy ALL copy C:\\root\\src\\project\\Windows\\wrapping\\python\\irtk.py C:\\root\\src\\project\\Windows\\lib\\release)
ENDIF (WIN32)
